package trycatchfinally;

public class TryCatchFinally {
public static void main(String[] args) {
int a,b = 5,c = 0;
try{
a = b/c;
System.out.println("Value of a is "+a);
}catch(ArithmeticException ae){
a = c/b;
System.out.println("Value of a is "+a);
}finally{
System.out.println("Finally block");
}
}
}

Output:

Value of a is 0
Finally block
