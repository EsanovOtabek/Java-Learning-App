package multilevelinheritence;

public class MultilevelInheritence {
public static void main(String[] args) {
C c = new C();
c.m3();
c.m1();
}
}
class A{
int c = 3;
void m1(){
System.out.println("I am in A");
}
}
class B extends A{
int a = 10;
void m2(){
System.out.println("I am in B");
}
}
class C extends B{
int b = a*c;
void m3(){
System.out.println("b = "+b);
m2();
}
}

Output:

b = 30
I am in B
I am in A
