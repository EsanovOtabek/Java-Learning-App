package fibonacci;

public class Fibonacci {
public static void main(String[] args) {
int n = 10,m = 0,l = 1,sum,i;
System.out.println("Fibonacci series is : ");
System.out.println(m);
System.out.println(l);
for(i = 1;i<=n;i++){
sum = m + l;
m = l;
l = sum;
System.out.println(sum);
}
}
}
