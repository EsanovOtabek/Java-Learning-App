package pkgfor;

public class For {
public static void main(String[] args) {
int[] arr = {12,42,90,77,56,81,11,62,40,55};
int sum = 0,i;
float average;
for(i=0;i < arr.length;i++){
sum = sum + arr[i];
}
average = (float)sum / arr.length;
System.out.println("Sum is "+sum);
System.out.println("Average is "+average);
}
}

Output:

Sum is 526
Average is 52.6
