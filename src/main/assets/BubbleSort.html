package bubblesort;

public class BubbleSort {
public static void main(String[] args) {
int[] data = {22,90,45,88,73};
int temp;
for(int i=0;i < data.length;i++){
for(int j=i+1;j < data.length;j++){
if(data[i] > data[j]){
temp = data[j];
data[j] = data[i];
data[i] = temp;
}
}
}
for(int d : data){
System.out.println(d);
}
}
}

Output:
22
45
73
88
90
