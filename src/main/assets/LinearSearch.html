package linear.search;

import java.util.Scanner;

public class LinearSearch {
public static void main(String[] args) {
Scanner s = new Scanner(System.in);
int arr[] = {27,64,43,90,42,88,12,138,45,33};
int item,i=0,flag=0;
System.out.println("Enter the item to be search in array");
item = s.nextInt();
for(i=0;i < arr.length;i++){
if(arr[i] == item){
System.out.println("Location of "+item+" is at "+i);
flag = 1;
}
}
if(flag == 0){
System.out.println("Element "+item+" is not present in array");
}
}
}

Output:

Enter the item to be search in array
45
Location of 45 is at 8
