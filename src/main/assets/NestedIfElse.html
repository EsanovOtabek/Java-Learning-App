package nestedif;

import java.util.Scanner;

public class NestedIf {
public static void main(String[] args) {
int m,n,o;
Scanner sc = new Scanner(System.in);
System.out.println("Enter the three integers");
m = sc.nextInt();
n = sc.nextInt();
o = sc.nextInt();
if(m > n){
if(m > o){
System.out.println("m = "+m+" is the largest number");
}else{
System.out.println("o = "+o+" is the largest number");
}
}
if(n > o){
System.out.println("n = "+n+" is the largest number");
}else{
System.out.println("o = "+o+" is the largest number");
}
}
}
