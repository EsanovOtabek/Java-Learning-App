package ojectasargument;

import java.util.Scanner;

public class OjectAsArgument {
public static void main(String[] args) {
System.out.println("Enter Time 1 in hours,minutes and seconds");
Time t1 = new Time();
Time t2 = new Time();
Time t3 = new Time();
t1.getTime();
System.out.println("Enter Time 2 in hours,minutes and seconds");
t2.getTime();
System.out.println("Addition of Time is");
t3 = t3.addTime(t1, t2);
System.out.println(t3.hours+":"+t3.minutes+":"+t3.seconds);
}
}
class Time{
int hours,minutes,seconds;
Scanner sc = new Scanner(System.in);
void getTime(){
hours = sc.nextInt();
minutes = sc.nextInt();
seconds = sc.nextInt();
}
Time addTime(Time tx1,Time tx2){
Time t = new Time();
t.hours = tx1.hours + tx2.hours + (tx1.minutes+tx2.minutes)/60;
t.minutes = (tx1.minutes + tx2.minutes)%60 + (tx1.seconds+tx2.seconds)/60;
t.seconds = (tx1.seconds + tx2.seconds)%60;
return t;
}
}
