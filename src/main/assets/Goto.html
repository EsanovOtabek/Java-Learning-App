package pkggoto;

public class Goto {
public static void main(String[] args) {
int data[] = {12,73,22,93,10,39,67,34,71,20};
int k = 0,max,loc;
max = data[0];
loc = 0;

check:
if(max<=data[k]){
max = data[k];
loc = k;
}

if(k < 9){
k++;
goto check;
}

System.out.println("Largest element : "+max);
System.out.println("Location of largest element : "+loc);
}
}

Output:

Largest element : 93
Location of largest element : 3
