package selectionsort;

public class SelectionSort {
public static void main(String[] args) {
int[] data = {22,90,45,88,73};
int min,temp;
for(int i=1;i < data.length;i++){
min = i;
for(int j=i+1;j < data.length;j++){
if(data[j] < data[min]){
temp = data[j];
data[j] = data[min];
data[min] = temp;
}
}
}
for(int d : data){
System.out.println(d);
}
}
}

Output:

22
45
73
88
90
