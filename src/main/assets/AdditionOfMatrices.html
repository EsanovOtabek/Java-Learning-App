package additionofmatrices;

public class AdditionOfMatrices {
public static void main(String[] args) {
int[][] a = {{5,6,9},{11,3,20},{47,11,10}};
int[][] b = {{23,14,66},{14,74,32},{10,14,21}};
int[][] c = new int[3][3];
int i,j;
System.out.println(Addition of two matrices is");
for(i=0;i < 3;i++){
for(j=0;j < 3;j++){
c[i][j] = a[i][j] + b[i][j];
}
}
for(i=0;i < 3;i++){
for(j=0;j < 3;j++){
System.out.print(c[i][j]+\t");
}
System.out.println();
}
}
}

Output:
Addition of two matrices is
28 20 75
25 77 52
57 25 31
