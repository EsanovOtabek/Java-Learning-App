package binarytodecimal;

public class BinaryToDecimal {
public static void main(String[] args) {
int b = 1100101,i = 0,r,sum = 0;
while(b!=0){
r = b % 10;
sum = (int) (sum + r*Math.pow(2,i));
i++;
b = b / 10;
}
System.out.println("Decimal equivalent of b = "+sum);
}
}

Output:
Decimal equivalent of b = 101
