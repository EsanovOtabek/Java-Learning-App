package insertionsort;

public class InsertionSort {
public static void main(String[] args) {
int[] data = {56,69,11,75,12};
int temp,j,i=1,k=1;
for(i=1;i < data.length;i++){
j = i - 1;
k = i;
while(j>=0){
if(data[j] > data[k]){
temp = data[j];
data[j] = data[k];
data[k] = temp;
}
j--;
k--;
}
}
for(int d : data){
System.out.println(d);
}
}
}

Output:
11
12
56
69
75
