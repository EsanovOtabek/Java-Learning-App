package simpleinheritence;

public class SimpleInheritence {
public static void main(String[] args) {
B b = new B();
b.m1();
b.m2();
}
}
class A{
int a = 5;
void m1(){
System.out.println("Hii I am m1");
}
}
class B extends A{
void m2(){
System.out.println("Value of a is "+a);
}
}

Output:

Hii I am m1
Value of a is 5
