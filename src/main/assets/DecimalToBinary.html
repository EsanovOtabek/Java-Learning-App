package decimaltobinary;

public class DecimalToBinary {
public static void main(String[] args) {
int d = 2356,r;
String s = "";
while(d!=0){
r = d % 2;
s = s + String.valueOf(r);
d = d/2;
} StringBuilder sb = new StringBuilder(s);
sb = sb.reverse();
System.out.println("Equivalent of d in binary is = "+sb);
}
}

Output:

Equivalent of d in binary is = 100100110100
