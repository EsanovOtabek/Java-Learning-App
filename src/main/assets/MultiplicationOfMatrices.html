package multiplicationofmatrices;

import java.util.Scanner;

public class MultiplicationOfMatrices {
public static void main(String[] args) {
Scanner sc = new Scanner(System.in);
int i,j,k,r1,c1,r2,c2,sum = 0;

System.out.println("Enter the no. of rows and columns for matrix 1");
r1 = sc.nextInt();
c1 = sc.nextInt();
System.out.println("Enter the no. of rows and columns for matrix 2");
r2 = sc.nextInt();
c2 = sc.nextInt();
if(c1 == r2){
int[][] a = new int[r1][c1];
int[][] b = new int[r2][c2];
int[][] m = new int[c1][r2];
System.out.println("Enter elements for matrix a");
for(i=0;i < r1;i++){
for(j=0;j < c1;j++){
a[i][j] = sc.nextInt();
}
}
System.out.println("Enter elements for matrix b");
for(i=0;i < r2;i++){
for(j=0;j < c2;j++){
b[i][j] = sc.nextInt();
}
}
System.out.println("The matrix a is :");
for(i=0;i < r1;i++){
for(j=0;j < c1;j++){
System.out.print(a[i][j]+"\t");
}
System.out.println();
}
System.out.println("The matrix b is :");
for(i=0;i < r2;i++){
for(j=0;j < c2;j++){
System.out.print(b[i][j]+"\t");
}
System.out.println();
}
System.out.println("Multiplication of matrices a and b is : ");
for(i=0;i < r1;i++){
for(j=0;j < c2;j++){
for(k=0;k < r2;k++){
sum = sum + a[i][k]*b[k][j];
}
m[i][j] = sum;
sum = 0;
}
}
for(i=0;i < r2;i++){
for(j=0;j < c1;j++){
System.out.print(m[i][j]+"\t");
}
System.out.println();
}
}else{
System.out.println("Matrix multiplication is not possible");
}
}
}
