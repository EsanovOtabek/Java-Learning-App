package onedarray;

public class OneDArray {
public static void main(String[] args) {
int[] a1 = {64,52,99,12,90};
int[] a2 = {56,12,88,32,82};
int[] a = new int[5];
for(int i=0;i < a1.length;i++){
a[i] = a1[i] + a2[i];
}
System.out.println("Addition of two 1D arrays is");
for(int arr:a){
System.out.println(arr);
}
}
}

Output:

Addition of two 1D arrays is
120
64
187
44
172
