package heirarichalinheritence;

public class HeirarichalInheritence {
public static void main(String[] args) {
B b = new B();
C c = new C();
c.m3();
c.m1();
b.m2();
b.m1();
}
}
class A{
int a = 0;
void m1(){
System.out.println("Value of a = " + a++);
}
}
class B extends A{
void m2(){
m1();
}
}
class C extends A{
void m3(){
m1();
}
}

Output:

Value of a = 0
Value of a = 1
Value of a = 0
Value of a = 1
